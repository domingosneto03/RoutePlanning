119vector<pair<int,int>> readSegments(const string &prompt) {
120 while (true) {
121 cout << prompt;
122 // clear leftover newline if present
123 if (cin.peek() == '\n') {
124 cin.ignore(numeric_limits<streamsize>::max(), '\n');
125 }
126
127 string line;
128 getline(cin, line);
129
130 if (line.empty()) {
131 // no segments
132 return {};
133 }
134
135 vector<pair<int,int>> segments;
136 bool valid = true;
137 stringstream ss(line);
138 string segment;
139
140 // e.g.: (1,2) (3,4)
141 while (ss >> segment) {
142 if (segment.front() != '(' || segment.back() != ')') {
143 cout << "Segment \"" << segment << "\" missing parentheses. Use (id1,id2)\n";
144 valid = false;
145 break;
146 }
147
148 // remove ( and )
149 segment = segment.substr(1, segment.size() - 2);
150
151 // parse inside
152 stringstream pairStream(segment);
153 string id1, id2;
154 if (!getline(pairStream, id1, ',') || !getline(pairStream, id2, ',')) {
155 cout << "Segment \"" << segment << "\" not in format id1,id2.\n";
156 valid = false;
157 break;
158 }
159
160 try {
161 int i1 = stoi(id1);
162 int i2 = stoi(id2);
163 segments.emplace_back(i1, i2);
164 } catch (...) {
165 cout << "Invalid integer in segment: (" << id1 << "," << id2 << ").\n";
166 valid = false;
167 break;
168 }
169 }
170
171 if (valid) {
172 return segments;
173 }
174 // otherwise, prompt again
175 }
176}
